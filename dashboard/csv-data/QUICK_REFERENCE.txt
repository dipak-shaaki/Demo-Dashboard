💾 HOW TO IMPORT TO DATABASE
==============================

MySQL Example:
--------------
LOAD DATA INFILE 'premium_collection.csv'
INTO TABLE premium_collection
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 ROWS;

PostgreSQL Example:
-------------------
COPY premium_collection FROM 'premium_collection.csv'
DELIMITER ',' CSV HEADER;


🐍 HOW TO USE IN PYTHON
=========================

import pandas as pd
import matplotlib.pyplot as plt

# Load data
premium = pd.read_csv('premium_collection.csv')
renewals = pd.read_csv('renewals_yearly.csv')

# Analysis
total_collection = premium['actual_amount'].sum()
avg_renewal_rate = renewals['renewal_rate_percent'].mean()

# Chart
premium.groupby('province')['actual_amount'].sum().plot(kind='bar')
plt.show()


📱 HOW TO USE IN JAVASCRIPT
=============================

// Load CSV
fetch('premium_collection.csv')
  .then(response => response.text())
  .then(csv => {
    // Parse CSV to JSON
    const lines = csv.split('\n');
    const headers = lines[0].split(',');
    const data = lines.slice(1).map(line => {
      const values = line.split(',');
      return headers.reduce((obj, header, i) => {
        obj[header] = values[i];
        return obj;
      }, {});
    });
    
    // Use in Chart.js
    createChart(data);
  });
